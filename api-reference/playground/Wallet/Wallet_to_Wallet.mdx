<ParamField body="fromAccount" type="string" required>

</ParamField>

<ParamField body="toAccount" type="string" required>

</ParamField>

<ParamField body="amount" type="number" required>

</ParamField>

<ParamField body="transactionReference" type="string" required>

</ParamField>

<ParamField body="remarks" type="string">
	Additional notes or description for the transaction. **Example:** "For

</ParamField>

<ParamField body="transactionTypeId" type="number">
	Unique identifier for the type of transaction. **Example:** 0
</ParamField>
