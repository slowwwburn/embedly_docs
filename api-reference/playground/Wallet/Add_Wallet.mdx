---
title: "Add Wallet"
api: "POST /wallets/add"
description: "The endpoint is designed to create a new wallet for a customer."
---

## Request Body

<ParamField body="id" type="string">
  Unique identifier for the wallet.
  **Example:** "3fa85f64-5717-4562-b3fc-2c963f66afa6"
</ParamField>

<ParamField body="walletGroupId" type="string">
  The ID of the wallet group this wallet belongs to.
  **Example:** "3fa85f64-5717-4562-b3fc-2c963f66afa6"
</ParamField>

<ParamField body="customerId" type="string" required>
  The ID of the customer who owns this wallet.
  **Example:** "3fa85f64-5717-4562-b3fc-2c963f66afa6"
</ParamField>

<ParamField body="availableBalance" type="number">
  The balance available for transactions.
  **Example:** 23444440
</ParamField>

<ParamField body="ledgerBalance" type="number">
  The total balance including pending transactions.
  **Example:** 500000
</ParamField>

<ParamField body="walletRestrictionId" type="string">
  ID referencing wallet restrictions.
  **Example:** "3fa85f64-5717-4562-b3fc-2c963f66afa6"
</ParamField>

<ParamField body="walletClassificationId" type="string">
  ID referencing wallet classification.
  **Example:** "3fa85f64-5717-4562-b3fc-2c963f66afa6"
</ParamField>

<ParamField body="currencyId" type="string" required>
  ID of the currency for this wallet.
  **Example:** "3fa85f64-5717-4562-b3fc-2c963f66afa6"
</ParamField>

<ParamField body="isInternal" type="boolean">
  Whether this is an internal wallet.
  **Example:** true
</ParamField>

<ParamField body="isDefault" type="boolean">
  Whether this is the default wallet for the customer.
  **Example:** true
</ParamField>

<ParamField body="name" type="string" required>
  Name of the wallet or wallet owner.
  **Example:** "Emmanuel Emah"
</ParamField>

<ParamField body="overdraft" type="number">
  Available overdraft amount, if any.
  **Example:** 0
</ParamField>

<ParamField body="virtualAccount" type="object">
  Virtual account details associated with the wallet.
</ParamField>

<Expandable title="virtualAccount properties">
  <ParamField body="virtualAccount.accountNumber" type="string">
    The virtual account number.
    **Example:** "0330000045"
  </ParamField>
  
  <ParamField body="virtualAccount.bankCode" type="string">
    The bank code for the virtual account.
    **Example:** "504"
  </ParamField>
  
  <ParamField body="virtualAccount.bankName" type="string">
    The name of the bank for the virtual account.
    **Example:** "Union"
  </ParamField>
</Expandable>

<ParamField body="mobNum" type="string">
  Mobile number associated with the wallet.
  **Example:** "2345678880"
</ParamField>

<ParamField body="customerTypeId" type="string">
  ID referencing the customer type.
  **Example:** "3fa85f64-5717-4562-b3fc-2c963f66afa6"
</ParamField>
