<ParamField body="CustomerId" type="string" required>
	The Embedly unique identifier of the customer whose KYC tier is being
	upgraded. **Example:** `3fa85f64-5717-4562-b3fc-2c963f66afa6`
</ParamField>

<ParamField body="UtilityBillDocument" type="file" required>
	The utility document of the customer's address.
</ParamField>

<ParamField body="BillType" type="string" required>
	The type of utility bill being provided. **Example:** `ELECTRICITY`. Posssible
	values: `ELECTRICITY | WATER | LAWMA`
</ParamField>

<ParamField body="Address.CustomerReference" type="string" required>
	The Embedly unique identifier of the customer whose KYC tier is being
	upgraded. **Example:** `3fa85f64-5717-4562-b3fc-2c963f66afa6`
</ParamField>

<ParamField body="Address.Street" type="string" required>
	The street where the customer resides
</ParamField>

<ParamField body="Address.LgaName" type="string" required>
	The local government where the customer resides
</ParamField>

<ParamField body="Address.StateName" type="string" required>
	The state where the customer resides
</ParamField>

<ParamField body="Address.Landmark" type="string" required>
	A Landmark of the area where the customer resides
</ParamField>

<ParamField body="Address.City" type="string" required>
	The city where the customer resides
</ParamField>

<ParamField body="Address.Firstname" type="string" required>
	The first name of the customer whose KYC tier is being upgraded.
</ParamField>

<ParamField body="Address.Lastname" type="string" required>
	The lastname of the customer whose KYC tier is being upgraded.
</ParamField>

<ParamField body="Address.Phone" type="string" required>
	The phone number of the customer whose KYC tier is being upgraded.
</ParamField>

<ParamField body="Address.Dob" type="string" required>
	The date of birth of the customer whose KYC tier is being upgraded.
</ParamField>

<ParamField body="Address.idType" type="string" required>
	The unique identification type of the customer whose KYC tier is being
	upgraded. **Example:** bvn. Possible values: `bvn | nin`
</ParamField>

<ParamField body="Address.idNumber" type="string" required>
	The unique identification number of the customer.
</ParamField>

<ParamField body="Address.Gender" type="string" required>
	The gender of the customer whose KYC tier is being upgraded. **Example:**
	Male. Possible values: `Male | Female`
</ParamField>

    			"BillType", "Address.StateName", "Address.City", "Address.Applicant.Lastname",
    			"Address.Applicant.Gender", "Address.Applicant.Phone", "Address.Applicant.IdNumber",

    			"Address.Applicant.Firstname", "Address.LgaName", "CustomerId",
    			"Address.Applicant.IdType", "Address.Street",
    	}

    	for _, field := range fields {
    			_ = writer.WriteField(field, "")
    	}

    	// Add file
    	file, _ := os.Open("path/to/utility_bill.pdf")
    	defer file.Close()
    	part, _ := writer.CreateFormFile("UtilityBillDocument", "utility_bill.pdf")
    	io.Copy(part, file)

    	writer.Close()

    	req, _ := http.NewRequest("POST", "https://waas-staging.embedly.ng/api/v1/customers/kyc/tier3-upgrade", body)
    	req.Header.Add("Content-Type", writer.FormDataContentType())

    	resp, _ := http.DefaultClient.Do(req)
    	defer resp.Body.Close()
    	respBody, _ := io.ReadAll(resp.Body)

    }
